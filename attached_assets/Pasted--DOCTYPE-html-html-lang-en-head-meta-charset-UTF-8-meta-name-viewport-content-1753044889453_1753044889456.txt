        function drawSecurityMesh(ctx) {
            const time = Date.now() * 0.001;
            const gridSize = 50;
            
            ctx.strokeStyle = getComputedStyle(document.documentElement)
                .getPropertyValue('--vault-cyan') + '20';
            ctx.lineWidth = 1;
            
            for (let x = 0; x < vaultMeshCanvas.width; x += gridSize) {
                for (let y = 0; y < vaultMeshCanvas.height; y += gridSize) {
                    const pulse = Math.sin(time + (x + y) * 0.01) * 0.5 + 0.5;
                    ctx.globalAlpha = pulse * 0.3;
                    
                    ctx.beginPath();
                    ctx.moveTo(x, y);
                    ctx.lineTo(x + gridSize, y);
                    ctx.lineTo(x + gridSize, y + gridSize);
                    ctx.stroke();
                }
            }
            ctx.globalAlpha = 1;
        }
        
        function drawBrandNodes(ctx) {
            nodes.forEach(node => {
                const pulse = Math.sin(Date.now() * 0.002 + node.id) * 0.3 + 0.7;
                
                ctx.fillStyle = node.sector === 'active' ? 
                    getComputedStyle(document.documentElement).getPropertyValue('--vault-cyan') :
                    getComputedStyle(document.documentElement).getPropertyValue('--muted-text') + '60';
                
                ctx.beginPath();
                ctx.arc(node.x, node.y, node.size * pulse, 0, Math.PI * 2);
                ctx.fill();
                
                // Add sector emoji
                ctx.font = `${node.size}px Arial`;
                ctx.textAlign = 'center';
                ctx.textBaseline = 'middle'; // Center vertically
                ctx.fillText(node.emoji, node.x, node.y);
            });
        }

        // --- Brand Element Display and Filtering ---
        function loadBrandElements(filterSectorKey = 'all') {
            const brandGrid = document.getElementById('brandGrid');
            brandGrid.innerHTML = ''; // Clear existing elements

            let filteredElements = allBrandElements;
            if (filterSectorKey !== 'all') {
                filteredElements = allBrandElements.filter(element => element.sector === filterSectorKey);
            }

            if (filteredElements.length === 0) {
                brandGrid.innerHTML = '<p class="text-center text-gray-500 col-span-full">No brand elements found for this selection.</p>';
                return;
            }

            filteredElements.forEach(element => {
                const card = document.createElement('div');
                card.classList.add('brand-element-card');
                card.innerHTML = `
                    <div class="element-name">${element.name}</div>
                    <div class="element-type">${element.type}</div>
                    <div class="element-sector">${element.sectorName}</div>
                `;
                brandGrid.appendChild(card);
            });
        }

            // Update active button state
            document.querySelectorAll('.sector-btn').forEach(btn => btn.classList.remove('active'));
            buttonElement.classList.add('active');

            loadBrandElements(sectorKey);
        }
